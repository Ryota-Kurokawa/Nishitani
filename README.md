# Day1

```ruby
# 変数の宣言と代入
x = 10

# 文字列
str = "Hello, World!"
puts str

# 配列
arr = [1, 2, 3, 4, 5]

# ハッシュ（キーと値のペア）
hash = {"key1" => "value1", "key2" => "value2"}

# 条件分岐
if x > 10
  puts "x is greater than 10"
elsif x == 10
  puts "x is equal to 10"
else
  puts "x is less than 10"
end

# ループ
5.times do
  puts "Hello, World!"
end

# 関数
def say_hello(name)
  puts "Hello, #{name}!"
end
```
